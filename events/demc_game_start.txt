namespace = demc_game_start

### Hidden Demon Cultists Game Start Event
event = {
	id = demc_game_start.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { has_ethic = ethic_demon_cultists }
			home_planet = {
				add_modifier = { modifier = demc_demon_gate }
			}
		}
	}
}
