button_pantheon_major_blood = {
	potential = {
		AND = {
			has_country_flag = pantheon_screen_open
			has_civic = civic_blood_demon_cult
		}
	}

	# You cannot stop following a major demon for 5 years after selecting one
	allow = {
		custom_tooltip = {
			text = PANTHEON_MAJOR_BLOOD_DESC
			NOT = {
				OR = {
					has_modifier = champion_of_the_blood_lord
					has_modifier = betrayer_of_the_blood_lord
				}
			}
		}
	}

	effect = {
		custom_tooltip = PANTHEON_MAJOR_BLOOD_DESC

		# If not active, enable and grant bonus
		if = {
			limit = { NOT = { has_modifier = champion_of_the_blood_lord_effect } }
			add_modifier = { modifier = champion_of_the_blood_lord_effect days = -1 }
			hidden_effect = {
				add_modifier = { modifier = champion_of_the_blood_lord days = 5 }
			}
			else = {
				hidden_effect = {
					remove_modifier = champion_of_the_blood_lord_effect
					add_modifier = { modifier = betrayer_of_the_blood_lord days = 5 }
				}
			}
		} 
	}
}
