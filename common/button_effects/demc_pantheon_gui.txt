	potential = {

		has_civic = civic_blood_demon_cult
	}

	# You cannot stop following a major demon for 5 years after selecting one

	}

	effect = {
		custom_tooltip = PANTHEON_MAJOR_BLOOD_DESC

		}
	}
}

button_pantheon_minor_blood_1 = {
	potential = {
		has_country_flag = pantheon_screen_open
		OR = {
			has_civic = civic_blood_demon_cult
			is_generic_cultist = yes
		}
		NOT = { has_modifier = minor_blood_demon_1_contract }
	}

	allow = {
		custom_tooltip = {
			text = PANTHEON_MINOR_BLOOD_1_DESC
		}
		custom_tooltip = {
			fail_text = PANTHEON_MINOR_CONTRACTED
			NOR = {
				has_modifier = minor_blood_demon_2_contract
				has_modifier = minor_blood_demon_3_contract
				has_modifier = minor_blood_demon_4_contract
				has_modifier = minor_blood_demon_5_contract
				has_modifier = minor_blood_demon_6_contract
			}
		}
		has_sacrificial_energy = yes
	}

	effect = {
		custom_tooltip = PANTHEON_MINOR_BLOOD_1_DESC
		change_variable = {
			which = sacrifical_energy_count
			value = -1
		}
		add_modifier = { modifier = minor_blood_demon_1_contract days = @contractTime }
		hidden_effect = {
			remove_country_flag = pantheon_minor_contract_canceled
			country_event = { id = demc_minor_demon.1 days = @contractNotificationTime }
		}
	}
}

button_pantheon_minor_blood_1_selected = {
	potential = {
		has_country_flag = pantheon_screen_open

	allow = {
		custom_tooltip = {
			text = PANTHEON_MINOR_BLOOD_1_DESC
		}

	effect = {
		custom_tooltip = PANTHEON_MINOR_BLOOD_1_DESC
		change_variable = {

		}
		remove_modifier = minor_blood_demon_1_contract

		hidden_effect = {
			set_country_flag = pantheon_minor_contract_canceled
		}
	}
}
