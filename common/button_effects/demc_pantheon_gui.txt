button_pantheon_screen_open = {
    potential = {
        check_variable = {
            which = "Root.pantheon_screen_open" 
            value = 0
        } 
    }
    
    allow = {
        always = yes
    }

    effect = {
        set_variable = {
            which = "Root.pantheon_screen_open" 
            value = 1
        }
    }
}

button_pantheon_major_blood = {
    potential = {
        AND = {
            check_variable = { which = "Root.pantheon_screen_open" value = 1 }
            has_civic = civic_blood_demon_cult
        }
    }

    # You cannot stop following a major demon for 5 years after selecting one
    allow = {
        NOT = {
            OR = {
                has_modifier = champion_of_the_blood_lord
                has_modifier = champion_of_the_plague_lord
                has_modifier = champion_of_the_pleasure_mistress
                has_modifier = champion_of_the_wise_one
            }
        }
    }

    effect = {
        remove_modifier = champion_of_the_plague_lord_effect
        remove_modifier = champion_of_the_pleasure_mistress_effect
        remove_modifier = champion_of_the_wise_one_effect
            
        add_modifier = { modifier = champion_of_the_blood_lord days = 1800 }
        add_modifier = { modifier = champion_of_the_blood_lord_effect days = -1 }
    }
}
