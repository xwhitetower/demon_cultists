pantheon_screen_open = {
    potential = {
        check_variable = {
            which = "Root.pantheon_screen_open" 
            value = 0
        } 
    }
    
    allow = {
        always = yes
    }

    effect = {
        set_variable = {
            which = "Root.pantheon_screen_open" 
            value = 1
        }
    }
}

    potential = {
        check_variable = {
            which = "Root.pantheon_screen_open" 
            value = 1
        }

        always = yes
    }

    effect = {
        set_variable = {
            which = "Root.pantheon_screen_open"
